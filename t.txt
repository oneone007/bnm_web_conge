<script>

const lampAnimation = document.getElementById('lampAnimation');
const htmlElement = document.documentElement;

// Load Lottie animation
let animation = lottie.loadAnimation({
    container: lampAnimation,
    renderer: 'svg',
    loop: false,  // Play only once per click
    autoplay: false, // Don't autoplay
    path: 'dark.json' // Ensure dark.json is in the correct directory
});

let isPlaying = false; // Track animation state

// Function to play animation briefly and stop
lampAnimation.addEventListener('click', () => {
    if (isPlaying) {
        animation.goToAndStop(0, true); // Reset animation to start
    } else {
        animation.play(); // Start animation

        // Stop animation after 1 second (adjust timing as needed)
        setTimeout(() => {
            animation.pause();
        }, 2000);
    }
    
    isPlaying = !isPlaying; // Toggle state

    // Toggle dark mode
    htmlElement.classList.toggle('dark');

    // Save theme preference
    localStorage.setItem('darkMode', htmlElement.classList.contains('dark'));
});

// Check for saved theme preference on page load
if (localStorage.getItem('darkMode') === 'true') {
    htmlElement.classList.add('dark');
} else {
    htmlElement.classList.remove('dark');
}


</script>
